Neave" date: "Febuary 2, 2017" --- var svg = d3.select("body") .append("svg") .attr("width", '800px') .attr("height", '800px') var m_data = [{x: 100, y:100}, {x: 200, y:200}, {x: 300, y: 300}] svg.selectAll("circle") .data(m_data) .enter().append("circle") .attr("cx", function(d) { return d.x; }) .attr("cy", function(d) { return d.y; }) .attr("r", 2.5);
